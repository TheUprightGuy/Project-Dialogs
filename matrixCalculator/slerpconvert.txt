#include <cmath>

void ConvertA();
{
GetA();

int w = QuatA[3];
int x = QuatA[0];
int y = QuatA[1];
int z = QuatA[2];

MatR[0][0] = 1 - 2 * pow(y, 2) - 2 * pow(z, 2);
MatR[0][1] = 2 * x * y - 2 * z * w;
MatR[0][2] = 2 * x * z + 2 * y * w;
MatR[0][3] = 0;

MatR[1][0] = 2 * x * y + 2 * z * w;
MatR[1][1] = 1 - 2 * pow(x, 2) - 2 * pow(z, 2);
MatR[1][2] = 2 * y * z - 2 * x * w;
MatR[1][3] = 0

MatR[2][0] = 2 * x * z - 2 * y * w;
MatR[2][1] = 2 * y * z + 2 * x * w;
MatR[2][2] = 1 - 2 * pow(x, 2) - 2 * pow(y, 2);
MatR[2][3] = 0

MatR[3][0] = 0
MatR[3][1] = 0
MatR[3][2] = 0
MatR[3][3] = 1

WriteToEditBox(hwnd, IDC_EDIT10, MatR[0][0]);
WriteToEditBox(hwnd, IDC_EDIT11, MatR[0][1]);
WriteToEditBox(hwnd, IDC_EDIT12, MatR[0][2]);
WriteToEditBox(hwnd, IDC_EDIT13, MatR[0][3]);

WriteToEditBox(hwnd, IDC_EDIT10, MatR[1][0]);
WriteToEditBox(hwnd, IDC_EDIT11, MatR[1][1]);
WriteToEditBox(hwnd, IDC_EDIT12, MatR[1][2]);
WriteToEditBox(hwnd, IDC_EDIT13, MatR[1][3]);

WriteToEditBox(hwnd, IDC_EDIT10, MatR[2][0]);
WriteToEditBox(hwnd, IDC_EDIT11, MatR[2][1]);
WriteToEditBox(hwnd, IDC_EDIT12, MatR[2][2]);
WriteToEditBox(hwnd, IDC_EDIT13, MatR[2][3]);

WriteToEditBox(hwnd, IDC_EDIT10, MatR[3][0]);
WriteToEditBox(hwnd, IDC_EDIT11, MatR[3][1]);
WriteToEditBox(hwnd, IDC_EDIT12, MatR[3][2]);
WriteToEditBox(hwnd, IDC_EDIT13, MatR[3][3]);
}
